#!/bin/bash

# =========================
# 1. Install Helm
# =========================
sudo apt-get update
sudo apt-get install -y curl gpg apt-transport-https

sudo mkdir -p /usr/share/keyrings

curl -fsSL https://packages.buildkite.com/helm-linux/helm-debian/gpgkey \
  | gpg --dearmor \
  | sudo tee /usr/share/keyrings/helm.gpg > /dev/null

echo "deb [signed-by=/usr/share/keyrings/helm.gpg] https://packages.buildkite.com/helm-linux/helm-debian/any/ any main" \
  | sudo tee /etc/apt/sources.list.d/helm-stable-debian.list

sudo apt-get update
sudo apt-get install -y helm

helm version

# =========================
# 2. Install ArgoCD
# =========================
helm repo add argo https://argoproj.github.io/argo-helm
helm repo update

helm install argocd argo/argo-cd \
  --namespace argocd \
  --create-namespace \
  --set server.service.type=NodePort

# =========================
# 3. Get ArgoCD NodePort
# =========================
echo "=== ArgoCD Service ==="
kubectl get svc -n argocd argocd-server

# =========================
# 4. Get ArgoCD Admin Password
# =========================
echo "=== ArgoCD Admin Password ==="
kubectl -n argocd get secret argocd-initial-admin-secret \
  -o jsonpath="{.data.password}" | base64 -d; echo

echo "======================================"
echo "Open your browser and go to:"
echo "https://<MASTER_IP>:<NODEPORT>"
echo "Username: admin"
echo "Password: (shown above)"
echo "======================================"

# =========================
# 5. Clone Config Repo
# =========================
git clone https://github.com/<your_username>/k8s-helm-config.git
cd k8s-helm-config

# =========================
# 6. Create Helm Chart
# =========================
helm create my-app

# =========================
# 7. Commit and Push
# =========================
git add .
git commit -m "Initial Helm chart commit"
git push origin main

# =========================
# 8. Create Namespace for App
# =========================
kubectl create namespace custom-app-ns

# =========================
# 9. Check Services (after ArgoCD deploys)
# =========================
echo "=== Services in custom-app-ns ==="
kubectl get svc -n custom-app-ns

echo "======================================"
echo "To test your app:"
echo "curl http://<MASTER_IP>:<APP_NODEPORT>"
echo "======================================"
